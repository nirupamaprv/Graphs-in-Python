# libraries
import pandas as pd
import numpy as np
import networkx as nx
import matplotlib.pyplot as plt
import warnings
warnings.filterwarnings("ignore")
 
# Build a dataframe with 4 connections
df = pd.DataFrame({ 'from':['A', 'B', 'C','A'], 'to':['D', 'A', 'E','C']})
df
 
# Build your graph
G=nx.from_pandas_dataframe(df, 'from', 'to')
 
# Plot it
nx.draw_networkx(G, with_labels=True)
plt.show()

# change node shape using parameter node_shape
# '^' gives triangle, '8' gives octagon, 'h' gives hexagon
# value needs to be one of ‘so^>v<dph8’ ; default=’o’ that is, circle.
#created dataframe from csv above
# create graph from above dataframe

Gq = nx.from_pandas_dataframe(dfx, 'NAMES', 'ACCOUNT')
plt.figure(3,figsize=(14,14)) #changes size of figure to 14 by 14
nx.draw_networkx(Gq, node_color='green', node_shape='^', with_labels=True)
plt.show()
